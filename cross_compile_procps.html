#!/bin/bash

CROSS_COMPILE=aarch64-none-linux-gnu-
ARCH=aarch64-none-linux-gnu
TARGET_DIR=$(cd `dirname $0`; pwd)/output
CACHE_FILE=aarch64.cache
build_procps() {
    cd procps-ng-3.3.11 && mkdir output
    echo "ac_cv_func_realloc_0_nonnull=yes" > $CACHE_FILE
    echo "ac_cv_func_malloc_0_nonnull=yes" >> $CACHE_FILE
    ./configure --host=$ARCH --prefix=$TARGET_DIR \
        --enable-static \
        --disable-shared \
        --without-ncurses \
        --cache-file=$CACHE_FILE \

        CC=$CROSS_COMPILE\gcc \
        STRIP=$CROSS_COMPILE\strip \
        RANLIB=$CROSS_COMPILE\ranlib \
        OBJDUMP=$CROSS_COMPILE\objdump \
        NM=$CROSS_COMPILE\nm \
        LD=$CROSS_COMPILE\ld \
        AR=$CROSS_COMPILE\ar \
        AS=$CROSS_COMPILE\as ;

    make ; make install || exit 1
    #make || exit 1
    cd -
}

build_procps
